# set global options for figures, code, warnings, and messages
knitr::opts_chunk$set(fig.width = 6, fig.height = 4, fig.path = "../figs/",
                      echo = FALSE, warning = FALSE, message = FALSE)
# load packages for current R session
library(dplyr)
library(readr)
library(ggplot2)
# ozone: import, select, drop missing observations, rename
ppm_and_time <- read_csv("C:/Users/Vance/OneDrive - Colostate/Desktop/Petersen---MECH476/data/ftc_o3.csv") %>% dplyr::select(sample_measurement, datetime) %>% filter(!is.na(sample_measurement), !is.na(datetime)) %>% rename(ozone_ppm = sample_measurement)
# examine dataframe object 
ppm_and_time
# create basic time series using ggplot2 package
ppm_and_time %>% ggplot() + geom_point(aes(x = datetime, y = ozone_ppm))
# create base layer of ozone time series (no geom) and save to object
ggplot_ozone <- ppm_and_time %>% ggplot(aes(x = datetime, y = ozone_ppm))
# add layer to ggplot object
ggplot_ozone + geom_point()
# add theme to ggplot object
ggplot_ozone + theme_light()
# create new object with ggplot labels 
ozone_labels <- labs(x = "Time of Measurement (YYYY-MM-DD HH:MM:SS)",
                     y = "Ozone Concentration (ppm)",
                     title = "Hourly Ozone Measurements in Fort Collins, CO")
# add title and axis labels to time series
ggplot_ozone + labs(x = "Time of Sample", y = "Ozone Concentration (ppm)", title = "Hourly Ozone Measurements in Fort Collins, CO")
# add alpha aesthetic to the geom_point()
ggplot_ozone + geom_point(alpha = 0.2)
# filter data to ozone concentration measurements exceeding 0.070 ppm 
ppm_and_time %>% filter(ozone_ppm > 0.07)
# time series of high ozone measurements
ppm_and_time %>% filter(ozone_ppm > 0.07) %>% ggplot(aes(x = datetime, y = ozone_ppm)) + geom_point(alpha = 0.4) + labs(x = "Time of Sample", y = "Ozone Concentration (ppm)", title = "Hourly Ozone Measurements in Fort Collins, CO", subtitle = "Exceeding EPA Regulations") + theme_light()
# calculate proportion of ozone measurements that exceed 0.070 ppm
paste0(round(sum(ppm_and_time$ozone_ppm > 0.07) / length(ppm_and_time$ozone_ppm) * 100, 2),"%")
